529  * Abstracts unique_ptr - exposes only void *get() method and a destructor.

533 struct unique_ptr_wrapper_base {
534  virtual ~unique_ptr_wrapper_base()
535  {

544  {
545  }
546 
547  void *get() override
548  {
549  return ptr.get();
550  }
551 
552  std::unique_ptr<T, D> ptr;

747 template <typename Comparator>
748 inline status config::put_comparator(Comparator &&comparator)
749 {
750  static_assert(
751  std::is_same<decltype(std::declval<Comparator>().compare(
752  std::declval<string_view>(),
753  std::declval<string_view>())),
754  int>::value,
755  "Comparator should implement `int compare(pmem::kv::string_view, pmem::kv::string_view)` method");
756  static_assert(std::is_convertible<decltype(std::declval<Comparator>().name()),
757  std::string>::value,
758  "Comparator should implement `std::string name()` method");
759 
760  std::unique_ptr<internal::comparator_base> wrapper;
761 
762  try {
763  wrapper = std::unique_ptr<internal::comparator_base>(
764  new internal::comparator_wrapper<Comparator>(
765  std::forward<Comparator>(comparator)));
766  } catch (std::bad_alloc &e) {
767  return status::OUT_OF_MEMORY;
768  } catch (...) {
769  return status::UNKNOWN_ERROR;
770  }
771 
772  auto cmp =
773  std::unique_ptr<pmemkv_comparator, decltype(pmemkv_comparator_delete) *>(
774  pmemkv_comparator_new(&internal::call_comparator_function,
775  std::string(comparator.name()).c_str(),
776  wrapper.get()),
777  &pmemkv_comparator_delete);
778  if (cmp == nullptr)
779  return status::UNKNOWN_ERROR;
780 
781  internal::unique_ptr_wrapper_base *entry;
782 
783  try {
784  entry = new internal::comparator_config_entry(std::move(wrapper),
785  std::move(cmp));
786  } catch (std::bad_alloc &e) {
787  return status::OUT_OF_MEMORY;
788  } catch (...) {
789  return status::UNKNOWN_ERROR;
790  }
791 
792  return static_cast<status>(pmemkv_config_put_object_cb(
793  this->config_.get(), "comparator", (void *)entry, internal::call_up_get,
794  internal::call_up_destructor));
795 }
