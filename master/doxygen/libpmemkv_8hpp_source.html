268 /*
269  * Abstracts unique_ptr - exposes only void *get() method and a destructor.
270  * This class is needed for C callbacks which cannot be templated
271  * (type of object and deleter must be abstracted away).
272  */
273 struct unique_ptr_wrapper_base {
274  virtual ~unique_ptr_wrapper_base()
275  {
276  }
277 
278  virtual void *get() = 0;
279 };
280 
281 template <typename T, typename D>
282 struct unique_ptr_wrapper : public unique_ptr_wrapper_base {
283  unique_ptr_wrapper(std::unique_ptr<T, D> ptr) : ptr(std::move(ptr))
284  {
285  }
286 
287  void *get() override
288  {
289  return ptr.get();
290  }
291 
292  std::unique_ptr<T, D> ptr;
293 };
294 
295 /*
296  * All functions which will be called by C code must be declared as extern "C"
297  * to ensure they have C linkage. It is needed because it is possible that
298  * C and C++ functions use different calling conventions.
299  */
300 extern "C" {
301 static inline void call_up_destructor(void *object)
302 {
303  auto *ptr = static_cast<unique_ptr_wrapper_base *>(object);
304  delete ptr;
305 }
306 
307 static inline void *call_up_get(void *object)
308 {
309  auto *ptr = static_cast<unique_ptr_wrapper_base *>(object);
310  return ptr->get();
311 }
312 } /* extern "C" */
313 } /* namespace internal */

437 {

439  return status::UNKNOWN_ERROR;
440

442
