280 /*
281  * Abstracts unique_ptr - exposes only void *get() method and a destructor.
282  * This class is needed for C callbacks which cannot be templated
283  * (type of object and deleter must be abstracted away).
284  */
285 struct unique_ptr_wrapper_base {
286  virtual ~unique_ptr_wrapper_base()
287  {
288  }
289 
290  virtual void *get() = 0;
291 };
292 
293 template <typename T, typename D>
294 struct unique_ptr_wrapper : public unique_ptr_wrapper_base {
295  unique_ptr_wrapper(std::unique_ptr<T, D> ptr) : ptr(std::move(ptr))
296  {
297  }
298 
299  void *get() override
300  {
301  return ptr.get();
302  }
303 
304  std::unique_ptr<T, D> ptr;
305 };

353  * to ensure they have C linkage. It is needed because it is possible that

535 template <typename Comparator>
536 inline status config::put_comparator(Comparator &&comparator)
537 {
538  static_assert(
539  std::is_same<decltype(std::declval<Comparator>().compare(
540  std::declval<string_view>(),
541  std::declval<string_view>())),
542  int>::value,
543  "Comparator should implement `int compare(pmem::kv::string_view, pmem::kv::string_view)` method");
544  static_assert(std::is_convertible<decltype(std::declval<Comparator>().name()),
545  std::string>::value,
546  "Comparator should implement `std::string name()` method");
547 
548  std::unique_ptr<internal::comparator_base> wrapper;
549 
550  try {
551  wrapper = std::unique_ptr<internal::comparator_base>(
552  new internal::comparator_wrapper<Comparator>(
553  std::forward<Comparator>(comparator)));
554  } catch (std::bad_alloc &e) {
555  return status::OUT_OF_MEMORY;
556  } catch (...) {
557  return status::UNKNOWN_ERROR;
558  }
559 
560  auto cmp =
561  std::unique_ptr<pmemkv_comparator, decltype(pmemkv_comparator_delete) *>(
562  pmemkv_comparator_new(&internal::call_comparator_function,
563  std::string(comparator.name()).c_str(),
564  wrapper.get()),
565  &pmemkv_comparator_delete);
566  if (cmp == nullptr)
567  return status::UNKNOWN_ERROR;
568 
569  internal::unique_ptr_wrapper_base *entry;
570 
571  try {
572  entry = new internal::comparator_config_entry(std::move(wrapper),
573  std::move(cmp));
574  } catch (std::bad_alloc &e) {
575  return status::OUT_OF_MEMORY;
576  } catch (...) {
577  return status::UNKNOWN_ERROR;
578  }
579 
580  return static_cast<status>(pmemkv_config_put_object_cb(
581  this->_config, "comparator", (void *)entry, internal::call_up_get,
582  internal::call_up_destructor));
583 }
