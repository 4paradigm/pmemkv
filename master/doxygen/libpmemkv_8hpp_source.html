283 /*
284  * Abstracts unique_ptr - exposes only void *get() method and a destructor.
285  * This class is needed for C callbacks which cannot be templated
286  * (type of object and deleter must be abstracted away).
287  */
288 struct unique_ptr_wrapper_base {
289  virtual ~unique_ptr_wrapper_base()
290  {
291  }
292 
293  virtual void *get() = 0;
294 };

296 template <typename T, typename D>
297 struct unique_ptr_wrapper : public unique_ptr_wrapper_base {
298  unique_ptr_wrapper(std::unique_ptr<T, D> ptr) : ptr(std::move(ptr))
299  {
300  }
301 
302  void *get() override
303  {
304  return ptr.get();
305  }
306 
307  std::unique_ptr<T, D> ptr;
308 };

538 template <typename Comparator>
539 inline status config::put_comparator(Comparator &&comparator)
540 {
541  static_assert(
542  std::is_same<decltype(std::declval<Comparator>().compare(
543  std::declval<string_view>(),
544  std::declval<string_view>())),
545  int>::value,
546  "Comparator should implement `int compare(pmem::kv::string_view, pmem::kv::string_view)` method");
547  static_assert(std::is_convertible<decltype(std::declval<Comparator>().name()),
548  std::string>::value,
549  "Comparator should implement `std::string name()` method");
550 
551  std::unique_ptr<internal::comparator_base> wrapper;
552 
553  try {
554  wrapper = std::unique_ptr<internal::comparator_base>(
555  new internal::comparator_wrapper<Comparator>(
556  std::forward<Comparator>(comparator)));
557  } catch (std::bad_alloc &e) {
558  return status::OUT_OF_MEMORY;
559  } catch (...) {
560  return status::UNKNOWN_ERROR;
561  }
562 
563  auto cmp =
564  std::unique_ptr<pmemkv_comparator, decltype(pmemkv_comparator_delete) *>(
565  pmemkv_comparator_new(&internal::call_comparator_function,
566  std::string(comparator.name()).c_str(),
567  wrapper.get()),
568  &pmemkv_comparator_delete);
569  if (cmp == nullptr)
570  return status::UNKNOWN_ERROR;
571 
572  internal::unique_ptr_wrapper_base *entry;
573 
574  try {
575  entry = new internal::comparator_config_entry(std::move(wrapper),
576  std::move(cmp));
577  } catch (std::bad_alloc &e) {
578  return status::OUT_OF_MEMORY;
579  } catch (...) {
580  return status::UNKNOWN_ERROR;
581  }
582 
583  return static_cast<status>(pmemkv_config_put_object_cb(
584  this->_config, "comparator", (void *)entry, internal::call_up_get,
585  internal::call_up_destructor));
586 }
