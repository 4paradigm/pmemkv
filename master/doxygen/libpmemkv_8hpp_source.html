579 /*
580  * Abstracts unique_ptr - exposes only void *get() method and a destructor.
581  * This class is needed for C callbacks which cannot be templated
582  * (type of object and deleter must be abstracted away).
583  */
584 struct unique_ptr_wrapper_base {
585  virtual ~unique_ptr_wrapper_base()
586  {
587  }
588 
589  virtual void *get() = 0;
590 };
591 
592 template <typename T, typename D>
593 struct unique_ptr_wrapper : public unique_ptr_wrapper_base {
594  unique_ptr_wrapper(std::unique_ptr<T, D> ptr) : ptr(std::move(ptr))
595  {
596  }
597 
598  void *get() override
599  {
600  return ptr.get();
601  }
602 
603  std::unique_ptr<T, D> ptr;
604 };

650 /*
651  * All functions which will be called by C code must be declared as extern "C"
652  * to ensure they have C linkage. It is needed because it is possible that
653  * C and C++ functions use different calling conventions.
654  */
655 extern "C" {
656 static inline void call_up_destructor(void *object)
657 {
658  auto *ptr = static_cast<unique_ptr_wrapper_base *>(object);
659  delete ptr;
660 }
661 
662 static inline void *call_up_get(void *object)
663 {
664  auto *ptr = static_cast<unique_ptr_wrapper_base *>(object);
665  return ptr->get();
666 }

669  size_t kb2, void *arg)

671  auto *cmp = static_cast<comparator_base *>(arg);
672  return cmp->compare(string_view(k1, kb1), string_view(k2, kb2));

798 template <typename Comparator>
799 inline status config::put_comparator(Comparator &&comparator)
800 {
801  static_assert(
802  std::is_same<decltype(std::declval<Comparator>().compare(
803  std::declval<string_view>(),
804  std::declval<string_view>())),
805  int>::value,
806  "Comparator should implement `int compare(pmem::kv::string_view, pmem::kv::string_view)` method");
807  static_assert(std::is_convertible<decltype(std::declval<Comparator>().name()),
808  std::string>::value,
809  "Comparator should implement `std::string name()` method");
810 
811  std::unique_ptr<internal::comparator_base> wrapper;
812 
813  try {
814  wrapper = std::unique_ptr<internal::comparator_base>(
815  new internal::comparator_wrapper<Comparator>(
816  std::forward<Comparator>(comparator)));
817  } catch (std::bad_alloc &e) {
818  return status::OUT_OF_MEMORY;
819  } catch (...) {
820  return status::UNKNOWN_ERROR;
821  }
822 
823  auto cmp =
824  std::unique_ptr<pmemkv_comparator, decltype(pmemkv_comparator_delete) *>(
825  pmemkv_comparator_new(&internal::call_comparator_function,
826  std::string(comparator.name()).c_str(),
827  wrapper.get()),
828  &pmemkv_comparator_delete);
829  if (cmp == nullptr)
830  return status::UNKNOWN_ERROR;
831 
832  internal::unique_ptr_wrapper_base *entry;
833 
834  try {
835  entry = new internal::comparator_config_entry(std::move(wrapper),
836  std::move(cmp));
837  } catch (std::bad_alloc &e) {
838  return status::OUT_OF_MEMORY;
839  } catch (...) {
840  return status::UNKNOWN_ERROR;
841  }
842 
843  return static_cast<status>(pmemkv_config_put_object_cb(
844  this->config_.get(), "comparator", (void *)entry, internal::call_up_get,
845  internal::call_up_destructor));
846 }
